### 注释

```less
// 这种注释不会被翻译成CSS
/*这种注册会被翻译成CSS*/
```

### 变量

```less
//定义变量的格式
@color:#ccc;
@w:4;
@h:4;
//@width:4;(×);
//@+变量名，变量名不能是CSS的关键字

//变量的作用域与延迟加载

@var:0;
.secect{
    @var:1
    .a{
       @var:2;
        three:@var;
        @var:3;
    }
    one:@var;
}

// one == 1;  three == 3;
//变量的作用域是块级作用域；延迟加载将在整个作用域结束后再去找变量，因此@var从2变成3
```

### 混合

```less
//普通混合会输出到css
.mini{
    
}

.a{
    .mini;//调用混合
}

//不输出到CSS
.mini(){
    
}

//带参数的混合
.mini(@w,@h,@c){
    
}

.a{
    .mini(4,4,ccc);
}

//带默认参数的混合
.mini(@w:10px,@h:10px,@c:#000){
    
}

.a{
    .mini();//有默认值，不传参也没事
}

//命名参数
.a{
    .mini(@c:ccc);//有默认值，不传参也没事
}

//匹配模式
.a(@_,@w){
    //带调用下列混合时，这个混合会被默认调用
}
.a(R,@w){
    
}
.a(L,@w){
    
}

.c{
    .a(R,400);//匹配模式
}

//arguments

.box(@a,@b,@c){
    border:@arguments;
}

.a{
    .box(1px,solid,#ccc);
}
```

```less
}

```
